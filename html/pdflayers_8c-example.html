pdflayers.c

This example shows how to create a PDF file containing layers; see also pretty printed pdflayers.c

// === Create a PDF file with layers
/*### Usage:
pdflayers PDF_FILE_NAME
*/
static const char usage_message[] =
"pdflayers - pdf create layer example\n"
"\n"
"usage:\n"
"\n"
" pdflayers PDF_FILE_NAME\n";
// ------------------------------------------------------------------------------
//### Includes
#include <stdio.h>
#include <stdlib.h>
#include "pdfrw.h"
#include "pdfannot.h"
// ------------------------------------------------------------------------------
//### Forwards
//Keep the examples bite-sized.

const char *fontName,

// ------------------------------------------------------------------------------
//### Print usage information and exit
/*
If something unexpected happens, we call this function and we're all done.
*/
static void usage(void)
{ fputs(usage_message, stderr);
exit(2);
}
// ------------------------------------------------------------------------------
//## Start here:
/*
`main` wraps `layers` in an error handler.
*/
int main(int argc, char **argv)
{
// drop the program name
argc--;
argv++;
// Verify one argument
if (argc != 1)
usage();
// Start up GCO
GCO_Init();
// Just in case: use a try block
ERR_TRY {
// Create a PDF
layers(argv[0]);
} ERR_CATCH {
// Something unexpected happened, report it
printf("Error %d: %s @ %s:%d\n",
ERR_NUMBER, ERR_MESSAGE, ERR_MODULE, ERR_LINE);
// Failure
GCO_Done();
return -1;
} ERR_END_CATCH
// Shut down GCO
GCO_Done();
// Success
return 0;
}
// ------------------------------------------------------------------------------
/*
#### Define some colors:
*/
static double red[3] = {1, 0, 0};
static double green[3] = {0, 1, 0};
static double blue[3] = {0, 0, 1};
// ------------------------------------------------------------------------------
//### Create a PDF file with layers
/*
This function creates a PDF file with some arbitrary contents.
Pdf is the high-level API; PdfDoc is the low-level API.
PdfRW is-a Pdf with write access enabled.
*/
void layers(const char *pdfFileName)
{
// opaque layer handles
void *layer1;
void *layer2;
void *layer3;
// Pdf object
Pdf *pdf;
// Create a PDF version 1.5 file:
pdf = (PdfPtr)PdfRW_Create(pdfFileName, 0x0105);
// If it didn't open, print the usage message and bail
if (!pdf)
usage();
// create the layers
layer1 = Pdf_CreateLayer(pdf, "First Layer (default on)", 1);
// example for a layer group
Pdf_CreateLayerTitle(pdf, "Layer Group");
// inside the group
layer2 = Pdf_CreateLayer(pdf, "Sub Layer (default on)", 1);
layer3 = Pdf_CreateLayer(pdf, "Another sub layer (default off)", 0);
/* Content within layers can nest, but that's confusing, so we're not going to do that
*/
// Create a new page
Pdf_BeginPage(pdf, 612, 792); // width and height
// inside First Layer
PDF_OpenLayer(pdf, layer1);
draw_text(pdf, "Helvetica", 12, red, 100, 700, "layer 1");
PDF_CloseLayer(pdf, layer1);
// inside Sub Layer
PDF_OpenLayer(pdf, layer2);
draw_text(pdf, "Helvetica", 12, green, 150, 650, "layer 2");
PDF_CloseLayer(pdf, layer2);
// inside Another sub Layer
PDF_OpenLayer(pdf, layer3);
draw_text(pdf, "Helvetica", 12, blue, 200, 600, "layer 3");
PDF_CloseLayer(pdf, layer3);
// Close the page
Pdf_EndPage(pdf);
// Discard closes the file and frees the object
GCO_Discard(pdf);
}
//
// ------------------------------------------------------------------------------
/*
### draw_text
Draw text using font size & color at x, y
*/
void draw_text(Pdf *pdf,
const char *fontName,
double fontSize,
double rgb[3],
double x,
double y,
const char *text)
{
void *font = Pdf_LoadFont(pdf, fontName, 0);
Pdf_SetFillRGB(pdf, rgb);
Pdf_SetFont(pdf, font, fontSize);
Pdf_SetTextPos(pdf, x, y);
Pdf_ShowText(pdf, text);
}
