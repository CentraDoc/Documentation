pdftype1font.c

This example shows how to create a PDF file that uses a type 1 font. see also pretty printed pdftype1font.c

// === PDF Type 1 font example
/*### Usage:
pdftype1font PDF_FILE_NAME
*/
static const char usage_message[] =
"pdftype1font - PDF Type 1 font example\n"
"\n"
"usage:\n"
"\n"
" pdftype1font PDF_FILE_NAME\n";
// ------------------------------------------------------------------------------
//### Includes
#include <stdio.h>
#include <stdlib.h>
#include "gcoreport.h"
#include "pdfrw.h"
#include "gx.h"
#include "type1gen.h"
// ------------------------------------------------------------------------------
//### Defines
#define NCHARS 6
//### Forwards
static void create_pdf(const char *fileName);
static void t1def(void *parm, int index, unsigned short *oChar, T1Gen *t1);
// ------------------------------------------------------------------------------
//### Print usage information and exit
/*
If something unexpected happens, we call this function and we're all done.
*/
static void usage(void)
{ fputs(usage_message, stderr);
exit(2);
}
// ------------------------------------------------------------------------------
//## Start here:
/*
`main` wraps `create_pdf` in an error handler.
*/
int main(int argc, char **argv)
{
// drop the program name
argc--;
argv++;
// verify one argument
if (argc != 1)
usage();
// start up GCO
GCO_Init();
// we need this for the memory leak report
GCO_KeepDebugInfo();
// just in case: use a try block
ERR_TRY {
// create the example PDF file
create_pdf(argv[0]);
} ERR_CATCH {
// something unexpected happened, report it
printf("Error %d: %s @ %s:%d\n",
ERR_NUMBER, ERR_MESSAGE, ERR_MODULE, ERR_LINE);
// failure
GCO_Done();
return -1;
} ERR_END_CATCH
// shut down GCO
GCO_Done();
// show the leak report
GCO_leakReport();
// success
return 0;
}
// ------------------------------------------------------------------------------
/*
#### Define some colors:
*/
static double red[3] = {1, 0, 0};
static double green[3] = {0, 1, 0};
static double blue[3] = {0, 0, 1};
// ------------------------------------------------------------------------------
//### Create a PDF file
/*
This function creates a PDF file demonstrating Type 1 fonts
page1 uses a narrow font
page2 uses a wide font
*/
static void page1(Pdf *pdf);
static void page2(Pdf *pdf);
void create_pdf(const char *fileName)
{
// create the PDF file
PdfPtr pdf = (PdfPtr)PdfRW_Create(fileName, 0x0105);
// oops
if (!pdf)
usage();
page1(pdf);
page2(pdf);
PDF_Discard(pdf);
}
static void page1(Pdf *pdf)
{
// create the font
void *t1 = PDF_DefineT1Font(pdf, "ExampleT1", NCHARS, 0, kFontEncWinANSI, t1def, 0);
// start a page
PDF_BeginPage(pdf, 612, 792);
PDF_SaveState(pdf);
// show the font in blue
PDF_SetFont(pdf, t1, 18);
PDF_SetFillRGB(pdf, blue);
// where?
PDF_SetTextPos(pdf, 20, 750);
// show
PDF_ShowText(pdf, "BACBe!");
// hey, we can use wide text with this narrow font!
PDF_ShowWideText(pdf, L"BACBe!");
// red underline manually
PDF_SetStrokeRGB(pdf, red);
PDF_MoveTo(pdf, 20, 750);
PDF_RLineTo(pdf, PDF_GetWideStringWidth(pdf, L"BACBe!"), 0);
PDF_Stroke(pdf);
PDF_RestoreState(pdf);
// manual kerning
{ char *pieces[] =
{
"B", "A", "C", "B", "e!"
};
unsigned short *wpieces[] =
{
L"B", L"A", L"C", L"B", L"e!"
};
double adjust[] =
{
-100, 600, -100, -300, 0
};
// italicize
double mtx[6] = { 1, 0, 0.15, 1, 0, 0 };
PDF_SaveState(pdf);
PDF_SetFont(pdf, t1, 18);
// red with blue outline
PDF_SetFillRGB(pdf, red);
PDF_SetStrokeRGB(pdf, blue);
PDF_SetTextRenderingMode(pdf, kTextRenderFillStroke);
// italic
PDF_SetTextMatrix(pdf, mtx);
// where?
PDF_SetTextPos(pdf, 20, 700);
// show
PDF_ShowTextAdjust(pdf, pieces, adjust, numberof(pieces));
// wide!
PDF_ShowWideTextAdjust(pdf, wpieces, adjust, numberof(pieces));
PDF_RestoreState(pdf);
}
// all done
PDF_EndPage(pdf);
}
static void page2(Pdf *pdf)
{
// create the font
void *t1 = PDF_DefineT1Font(pdf, "WideUnicodeT1", NCHARS, 0, kFontUseWideEncoding + kFontEncUnicode, t1def, 0);
// start a page
PDF_BeginPage(pdf, 612, 792);
PDF_SaveState(pdf);
// show the font in blue
PDF_SetFont(pdf, t1, 18);
PDF_SetFillRGB(pdf, blue);
// where?
PDF_SetTextPos(pdf, 20, 750);
// what?
PDF_ShowWideText(pdf, L"BACBe!");
// NOTE: We can pass Wide text to the narrow ShowText call(!) (legacy)
// but we can't pass narrow text for this font!
PDF_ShowText(pdf, (const char *)L" BACBe!");
// red underline (the first part) manually
PDF_SetStrokeRGB(pdf, red);
PDF_MoveTo(pdf, 20, 750);
PDF_RLineTo(pdf, PDF_GetWideStringWidth(pdf, L"BACBe!"), 0);
PDF_Stroke(pdf);
PDF_RestoreState(pdf);
// manual kerning
{ unsigned short *pieces[] =
{
L"B", L"A", L"C", L"B", L"e!"
};
double adjust[] =
{
-100, 600, -100, -300, 0
};
// italicize
double mtx[6] = { 1, 0, 0.15, 1, 0, 0 };
PDF_SaveState(pdf);
PDF_SetFont(pdf, t1, 18);
// red with blue outline
PDF_SetFillRGB(pdf, red);
PDF_SetStrokeRGB(pdf, blue);
PDF_SetTextRenderingMode(pdf, kTextRenderFillStroke);
// italic
PDF_SetTextMatrix(pdf, mtx);
// where?
PDF_SetTextPos(pdf, 20, 700);
// show w/adjustments
PDF_ShowWideTextAdjust(pdf, pieces, adjust, numberof(pieces));
// NOTE: We can also pass Wide text to the narrow ShowTextAdjust(!) (legacy)
// but we can't pass narrow text for this font!
PDF_ShowTextAdjust(pdf, (const char **)pieces, adjust, numberof(pieces));
PDF_RestoreState(pdf);
}
// all done
PDF_EndPage(pdf);
}

/*
#### Define a small Type 1 font with
This function is called from inside `PDF_DefineT1Font` for `index = 0 to N`.
Type 1 functions define a .notdef glyph 0
The drawing interface is defined in gfx/t1gen.h -
similar to the PDF or PostScript model, with some differences.
*/
void t1def(void *parm, int index, unsigned short *oChar, T1Gen *t1)
{
// requested glyph index:
switch (index) {
// left triangle
case 1:
*oChar = 'A';
T1Gen_BeginChar(t1, index, 750);
T1Gen_MoveTo(t1, 0, 0);
T1Gen_LineTo(t1, 0, 1000);
T1Gen_LineTo(t1, 700, 0);
T1Gen_ClosePath(t1);
T1Gen_EndChar(t1);
break;
// solid box
case 2:
*oChar = 'B';
T1Gen_BeginChar(t1, index, 500);
T1Gen_MoveTo(t1, 0, 0);
T1Gen_LineTo(t1, 0, 1000);
T1Gen_LineTo(t1, 450, 1000);
T1Gen_LineTo(t1, 450, 0);
T1Gen_ClosePath(t1);
T1Gen_EndChar(t1);
break;
case 3:
// right triangle
*oChar = 'C';
T1Gen_BeginChar(t1, index, 750);
T1Gen_MoveTo(t1, 0, 1000);
T1Gen_LineTo(t1, 700, 0);
T1Gen_LineTo(t1, 700, 1000);
T1Gen_ClosePath(t1);
T1Gen_EndChar(t1);
break;
case 4:
// lower case e
*oChar = 'e';
/* e */
T1Gen_BeginChar(t1, index, 556);
T1Gen_RMoveTo(t1, 516, 236);
T1Gen_LineTo(t1, 130, 236);
T1Gen_CurveTo(t1, 132, 126, 191, 64, 287, 64);
T1Gen_CurveTo(t1, 359, 64, 408, 101, 423, 169);
T1Gen_RLineTo(t1, 84, 0);
T1Gen_RRCurveTo(t1, -18, -115, -87, -69, -129, 0);
T1Gen_RRCurveTo(t1, -141, 0, -92, 100, 0, 176);
T1Gen_RRCurveTo(t1, 0, 174, 93, 103, 144, 0);
T1Gen_RRCurveTo(t1, 151, 0, 88, -94,0, -170);
T1Gen_RRCurveTo(t1, 0, -11, 0, -14, 0, -13);
T1Gen_ClosePath(t1);
T1Gen_RMoveTo(t1, -91, 73);
T1Gen_RRCurveTo(t1, -2, 98, -52, 53, -93, 0);
T1Gen_RRCurveTo(t1, -87, 0, -53, -53, -8, -98);
T1Gen_RLineTo(t1, 295, 0);
T1Gen_ClosePath(t1);
T1Gen_EndChar(t1);
break;
case 5:
// an actual exclamation point
*oChar = '!';
/* ! */
T1Gen_BeginChar(t1, index, 278);
T1Gen_RMoveTo(t1, 90, 107);
T1Gen_RLineTo(t1, 97, 0);
T1Gen_RLineTo(t1, 0, -107);
T1Gen_RLineTo(t1, -97, 0);
T1Gen_RLineTo(t1, 0, 107);
T1Gen_RMoveTo(t1, 0, 376);
T1Gen_RLineTo(t1, 0, 235);
T1Gen_RLineTo(t1, 97, 0);
T1Gen_RLineTo(t1, 0, -235);
T1Gen_RLineTo(t1, -25, -310);
T1Gen_RLineTo(t1, -50, 0);
T1Gen_RLineTo(t1, -22, 310);
T1Gen_ClosePath(t1);
T1Gen_EndChar(t1);
break;
// nul char
default:
*oChar = index;
T1Gen_BeginChar(t1, index, 750);
T1Gen_EndChar(t1);
break;
}
}
