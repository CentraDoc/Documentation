This example shows how to create a PDF file using the high level interface; see also pretty printed pdfcreate1.c

// === Create a PDF file using the high level interface
/*### Usage:
pdfcreate1 PDF_FILE_NAME
*/
static const char usage_message[] =
"pdfcreate1 - pdf create example\n"
"\n"
"usage:\n"
"\n"
" pdfcreate1 PDF_FILE_NAME\n";
// ------------------------------------------------------------------------------
//### Includes
#include <stdio.h>
#include <stdlib.h>
#include "pdfrw.h"
#include "pdfannot.h"
// ------------------------------------------------------------------------------
//### Forwards
//Keep the examples bite-sized.
static void create1(const char *pdfFileName);
static void draw_text(Pdf *pdf,
const char *fontName,
double size,
double rgb[3],
double x,
double y,
const char *text);
static void rotate_example(Pdf *pdf);
static void path_example(Pdf *pdf);
// ------------------------------------------------------------------------------
//### Print usage information and exit
/*
If something unexpected happens, we call this function and we're all done.
*/
static void usage(void)
{ fputs(usage_message, stderr);
exit(2);
}
// ------------------------------------------------------------------------------
//## Start here:
/*
`main` wraps `create1` in an error handler.
*/
int main(int argc, char **argv)
{
// drop the program name
argc--;
argv++;
// Verify one argument
if (argc != 1)
usage();
// Start up GCO
GCO_Init();
// Just in case: use a try block
ERR_TRY {
// Create a PDF
create1(argv[0]);
} ERR_CATCH {
// Something unexpected happened, report it
printf("Error %d: %s @ %s:%d\n",
ERR_NUMBER, ERR_MESSAGE, ERR_MODULE, ERR_LINE);
// Failure
GCO_Done();
return -1;
} ERR_END_CATCH
// Shut down GCO
GCO_Done();
// Success
return 0;
}
/*
This alternative closing sequence will report memory leaks:
// #include "gcoreport.h" required for GCO_leakReport();
GCO_KeepDebugInfo();
GCO_Done();
GCO_leakReport();
However: if the program calls:
GCO_Init();
//...
GCO_Done();
and then calls `GCO_Init();` again later, do not use `GCO_KeepDebugInfo();` or
bad things can happen. `GCO_KeepDebugInfo();` short circuits some teardown
logic.
*/
// ------------------------------------------------------------------------------
/*
#### Define some colors:
These are the basic Gray, RGB and CMYK colorspace colors.
*/
static double black = 0;
static double white = 1;
static double gray = 0.5;
static double red[3] = {1, 0, 0};
static double green[3] = {0, 1, 0};
static double blue[3] = {0, 0, 1};
static double cyan[4] = {1, 0, 0, 0};
static double magenta[4] = {0, 1, 0, 0};
static double yellow[4] = {0, 0, 1, 0};
// ------------------------------------------------------------------------------
//### Create a PDF file
/*
This function creates a PDF file with some arbitrary contents.
Pdf is the high-level API; PdfDoc is the low-level API.
PdfRW is-a Pdf with write access enabled.
*/
void create1(const char *pdfFileName)
{
Pdf *pdf;
// Create a PDF version 1.5 file:
pdf = (PdfPtr)PdfRW_Create(pdfFileName, 0x0105);
// If it didn't open, print the usage message and bail
if (!pdf)
usage();
// Create a new page
Pdf_BeginPage(pdf, 612, 792); // width and height
//### Examples
// Add a bookmark at the top of this page
Pdf_AddBookmark(pdf, "FirstPage", kNodeClosed, kNodeSameLevel, 0, 792);
// hello world
draw_text(pdf, "Helvetica", 12, blue, 100, 700, "hello world");
// rotated text
rotate_example(pdf);
// fill and stroke path
path_example(pdf);
// Close the page
Pdf_EndPage(pdf);
// Discard closes the file and frees the object
GCO_Discard(pdf);
}
//
// ------------------------------------------------------------------------------
/*
### draw_text
Draw text using font size & color at x, y
*/
void draw_text(Pdf *pdf,
const char *fontName,
double fontSize,
double rgb[3],
double x,
double y,
const char *text)
{
void *font = Pdf_LoadFont(pdf, fontName, 0);
Pdf_SetFillRGB(pdf, rgb);
Pdf_SetFont(pdf, font, fontSize);
Pdf_SetTextPos(pdf, x, y);
Pdf_ShowText(pdf, text);
}
// ------------------------------------------------------------------------------
//### rotate_example
/* hello world transformed: rotate and squeeze. Use `Pdf_SaveState` and
`Pdf_RestoreState` to temporarily change graphic state information.
`Pdf_SaveState` and `Pdf_RestoreState` are *required* when using clipping; there
is a function `Pdf_SetClip` but no `Pdf_RemoveClip`, it's just not part of the PDF
model.
*/
static void rotate_example(Pdf *pdf)
{
Pdf_SaveState(pdf);
Pdf_Translate(pdf, 100, 700);
Pdf_Rotate(pdf, 30);
Pdf_Scale(pdf, .5, 1);
draw_text(pdf, "Helvetica", 12, red, 0, 0, "hello world");
Pdf_RestoreState(pdf);
}
// ------------------------------------------------------------------------------
//### path_example
/*
Arbitrary paths; fill and/or stroke
*/
static void path_example(Pdf *pdf)
{
Pdf_MoveTo(pdf, 100, 690);
Pdf_LineTo(pdf, 150, 690);
/* Arc ClockWise:
*/
// `Pdf_ArcCW(centerX, centerY, radius, startAngle, endAngle);`
Pdf_ArcCW(pdf, 100, 690, 50, 0, 270);
Pdf_SetFillCMYK(pdf, cyan);
Pdf_SetStrokeRGB(pdf, blue);
Pdf_SetLineWidth(pdf, 3);
Pdf_CloseFillStroke(pdf);
}
