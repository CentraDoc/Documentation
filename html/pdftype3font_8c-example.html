pdftype3font.c

This example shows how to create a PDF file that uses a type 3 font. see also pretty printed pdftype3font.c

// === PDF Type 3 font example
/*### Usage:
pdftype3font PDF_FILE_NAME
*/
static const char usage_message[] =
"pdftype3font - PDF Type 3 font example\n"
"\n"
"usage:\n"
"\n"
" pdftype3font PDF_FILE_NAME\n";
// ------------------------------------------------------------------------------
//### Includes
#include <stdio.h>
#include <stdlib.h>
#include "gcoreport.h"
#include "pdfrw.h"
#include "gx.h"
// ------------------------------------------------------------------------------
//### Forwards
static void create_pdf(const char *fileName);
static void t3stencildef(void *parm, int index, unsigned short *oChar, Pdf *pdf);
static void t3colordef(void *parm, int index, unsigned short *oChar, Pdf *pdf);
// ------------------------------------------------------------------------------
//### Print usage information and exit
/*
If something unexpected happens, we call this function and we're all done.
*/
static void usage(void)
{ fputs(usage_message, stderr);
exit(2);
}
// ------------------------------------------------------------------------------
//## Start here:
/*
`main` wraps `create_pdf` in an error handler.
*/
int main(int argc, char **argv)
{
// drop the program name
argc--;
argv++;
// verify one argument
if (argc != 1)
usage();
// start up GCO
GCO_Init();
// we need this for the memory leak report
GCO_KeepDebugInfo();
// just in case: use a try block
ERR_TRY {
// create the example PDF file
create_pdf(argv[0]);
} ERR_CATCH {
// something unexpected happened, report it
printf("Error %d: %s @ %s:%d\n",
ERR_NUMBER, ERR_MESSAGE, ERR_MODULE, ERR_LINE);
// failure
GCO_Done();
return -1;
} ERR_END_CATCH
// shut down GCO
GCO_Done();
// show the leak report
GCO_leakReport();
// success
return 0;
}
// ------------------------------------------------------------------------------
/*
#### Define some colors:
*/
static double red[3] = {1, 0, 0};
static double green[3] = {0, 1, 0};
static double blue[3] = {0, 0, 1};
// ------------------------------------------------------------------------------
//### Create a PDF file
/*
This function creates a PDF file demonstrating Type 3 fonts
*/
void create_pdf(const char *fileName)
{
void *t3stencil;
void *t3color;
// create the PDF file
PdfPtr pdf = (PdfPtr)PdfRW_Create(fileName, 0x0105);
// oops
if (!pdf)
usage();
// create the stencil font
t3stencil = PDF_DefineT3Font(pdf, "Alpha", 3, 0, 1, t3stencildef, 0);
// create the color font
t3color = PDF_DefineT3Font(pdf, "Beta", 3, 0, 1, t3colordef, 0);
// start a page
PDF_BeginPage(pdf, 612, 792);
// show the stencil font in blue
PDF_SetFont(pdf, t3stencil, 18);
// note that stroke color does not affect stencil font rendering!
PDF_SetFillRGB(pdf, blue);
// where?
PDF_SetTextPos(pdf, 100, 700);
// what?
PDF_ShowText(pdf, "ABCABC");
// show the color font
PDF_SetFont(pdf, t3color, 18);
// where?
PDF_SetTextPos(pdf, 100, 650);
// what?
PDF_ShowText(pdf, "CBACBA");
//#### Manual kerning and clip rendering
{
// `PDF_ShowTextAdjust` draws multiple fragments with position adjustments
char *pieces[] = { "BA", "CBA", "CB" };
// adjustments are in glyph space and are *subtracted* from the next position
double adjust[] = { 300, 300, 0 };
// saves the current clip (which is none)
PDF_SaveState(pdf);
// Must use the stencil font
PDF_SetFont(pdf, t3stencil, 18);
// where?
PDF_SetTextPos(pdf, 100, 600);
/*
Use the text to set a clip region but don't draw anything.
Note that this mode does not work with a color type 3 font!
*/
PDF_SetTextRenderingMode(pdf, kTextRenderNoneAddClip);
PDF_ShowTextAdjust(pdf, pieces, adjust, numberof(pieces));
// fill a rectangle (anything drawn will clip to the text above)
PDF_Rectangle(pdf, 100, 600, 100, 50);
PDF_SetFillRGB(pdf, green);
PDF_Fill(pdf);
// restores the clip (back to none)
PDF_RestoreState(pdf);
}
// all done
PDF_EndPage(pdf);
PDF_Discard(pdf);
}
// ------------------------------------------------------------------------------
/*
#### Define a small Type 3 stencil font with 3 glyphs
This function is called from inside `PDF_DefineT3Font` for `index = 0 to n-1`.
Type 3 functions draw using the PDF API and must return the encoding value for the character.
A Type3 stencil glyph definition must begin with `PDF_BeginStencilT3Glyph`.
Stencil definitions must not call any color setting functions.
*/
static void t3stencildef(void *parm, int index, unsigned short *oChar, Pdf *pdf)
{
// used for `PDF_BeginStencilT3Glyph`
int bbox[4] = { 0, 0, 1000, 1000 };
// requested glyph index:
switch (index) {
// solid triangle pointing up
case 0:
*oChar = 'A';
bbox[2] = 750;
PDF_BeginStencilT3Glyph(pdf, 750, bbox);
PDF_MoveTo(pdf, 0, 0);
PDF_LineTo(pdf, 350, 1000);
PDF_LineTo(pdf, 700, 0);
PDF_ClosePath(pdf);
PDF_Fill(pdf);
break;
// box frame
case 1:
*oChar = 'B';
bbox[2] = 500;
PDF_BeginStencilT3Glyph(pdf, 500, bbox);
PDF_SetLineWidth(pdf, 50);
PDF_MoveTo(pdf, 25, 25);
PDF_LineTo(pdf, 25, 975);
PDF_LineTo(pdf, 450, 975);
PDF_LineTo(pdf, 450, 25);
PDF_ClosePath(pdf);
PDF_Stroke(pdf);
break;
// solid triangle pointing down
case 2:
*oChar = 'C';
bbox[2] = 750;
PDF_BeginStencilT3Glyph(pdf, 750, bbox);
PDF_MoveTo(pdf, 0, 1000);
PDF_LineTo(pdf, 350, 0);
PDF_LineTo(pdf, 700, 1000);
PDF_ClosePath(pdf);
PDF_Fill(pdf);
break;
}
}
// ------------------------------------------------------------------------------
/*
#### Define a small Type 3 color font with (the same) 3 glyphs in color
This function is called from inside `PDF_DefineT3Font` for `index = 0 to n-1`.
Type 3 functions draw using the PDF API and must return the encoding value for the character.
A Type 3 color glyph definition must begin with `PDF_BeginColorT3Glyph`.
Color definitions must set the color before drawing.
*/
static void t3colordef(void *parm, int index, unsigned short *oChar, Pdf *pdf)
{
// requested glyph index:
switch (index) {
// solid blue triangle pointing up
case 0:
*oChar = 'A';
PDF_BeginColorT3Glyph(pdf, 750);
PDF_SetFillRGB(pdf, blue);
PDF_MoveTo(pdf, 0, 0);
PDF_LineTo(pdf, 350, 1000);
PDF_LineTo(pdf, 700, 0);
PDF_ClosePath(pdf);
PDF_Fill(pdf);
break;
// red box frame
case 1:
*oChar = 'B';
PDF_BeginColorT3Glyph(pdf, 500);
PDF_SetStrokeRGB(pdf, red);
PDF_SetLineWidth(pdf, 50);
PDF_MoveTo(pdf, 25, 25);
PDF_LineTo(pdf, 25, 975);
PDF_LineTo(pdf, 450, 975);
PDF_LineTo(pdf, 450, 25);
PDF_ClosePath(pdf);
PDF_Stroke(pdf);
break;
// red triangle with blue frame
case 2:
*oChar = 'C';
PDF_BeginColorT3Glyph(pdf, 750);
PDF_SetFillRGB(pdf, red);
PDF_SetStrokeRGB(pdf, blue);
PDF_SetLineWidth(pdf, 50);
PDF_MoveTo(pdf, 25, 975);
PDF_LineTo(pdf, 350, 25);
PDF_LineTo(pdf, 675, 975);
PDF_ClosePath(pdf);
PDF_FillStroke(pdf);
break;
}
}
